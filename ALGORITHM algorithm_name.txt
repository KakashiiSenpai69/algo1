ALGORITHM algorithm_name
VAR
   ch :  STRING;
   nbv,nbw,i: INTEGER 
BEGIN
    nbw:=1
    nbv:=0
    i:=1
    Write("give ch")
    Read(ch)
    WHILE (ch[i]<>".") DO
        IF (ch[i] in ["a","e","o","u","i","y"]) THEN
            nbv:=nbv+1
        END_IF
        IF (ch[i]=" ") THEN
            nbw:=nbw+1
        END_IF
        i:=i+1
    END_WHILE
    Write("num of words",nbw)
    Write("num of  voyels",nbv)
    Write("length",i+1)

    
END
